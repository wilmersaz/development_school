function crearProducto(nombre, precio, categoria, opciones = {}) {
  // Configuraci√≥n por defecto basada en categor√≠a

  };

  return {
    id: `PROD-${Date.now()}-${Math.random().toString(36).substr(2, 4).toUpperCase()}`,

const configSistema = {
  nombre: "MiTienda Online",

// Factory para crear tienda din√°mica
function crearTienda(configuracion = {}) {
  const config = { ...configSistema, ...configuracion };
  
  return {
    ...config,
    productos: [],
    usuarios: [],
    pedidos: [],
    
    // M√©todos din√°micos
    agregarProducto(datosProducto) {
      const producto = {
        id: `PROD-${Date.now()}`,
        ...datosProducto,
        fechaCreacion: new Date().toISOString(),
        activo: true,
        
        // M√©todo para calcular precio con impuestos
        precioConImpuestos() {
          return this.precio * (1 + config.impuesto);
        }
      };
      
      this.productos.push(producto);
      return producto;
    },
    
    crearUsuario(datosUsuario, tipoSuscripcion = "regular") {
      const beneficios = {
        regular: { descuento: 0, envioGratis: config.envioGratis },
        premium: { descuento: 0.1, envioGratis: 25 },
        vip: { descuento: 0.15, envioGratis: 0 }
      };
      
      const usuario = {
        id: `USER-${Date.now()}`,
        ...datosUsuario,
        tipo: tipoSuscripcion,
        ...beneficios[tipoSuscripcion],
        fechaRegistro: new Date().toISOString(),
        pedidos: [],
        
        realizarPedido(productosIds, metodoPago) {
          const productosEncontrados = this.productos.filter(p => 
            productosIds.includes(p.id)
          );
          
          if (productosEncontrados.length === 0) {
            return { error: "No se encontraron productos v√°lidos" };
          }
          
          const subtotal = productosEncontrados.reduce((sum, p) => 
            sum + p.precioConImpuestos(), 0
          );
          
          const descuentoAplicado = subtotal * this.descuento;
          const totalConDescuento = subtotal - descuentoAplicado;
          
          const costoEnvio = totalConDescuento >= this.envioGratis ? 0 : 10;
          const total = totalConDescuento + costoEnvio;
          
          const comisionPago = config.configuracionPagos[metodoPago]?.comision || 0;
          const comision = total * comisionPago;
          
          const pedido = {
            id: `PED-${Date.now()}`,
            usuarioId: this.id,
            productos: productosEncontrados.map(p => ({
              id: p.id,
              nombre: p.nombre,
              precio: p.precio,
              precioConImpuestos: p.precioConImpuestos()
            })),
            subtotal,
            descuento: descuentoAplicado,
            envio: costoEnvio,
            comision,
            total: total + comision,
            metodoPago,
            fecha: new Date().toISOString(),
            estado: "pendiente"
          };
          
          this.pedidos.push(pedido);
          usuario.pedidos.push(pedido.id);
          
          return pedido;
        }.bind(this)
      };
      
      this.usuarios.push(usuario);
      return usuario;
    },
    
    obtenerEstadisticas() {
      const totalProductos = this.productos.length;
      const totalUsuarios = this.usuarios.length;
      const totalPedidos = this.pedidos.length;
      
      const ingresoTotal = this.pedidos.reduce((sum, p) => sum + p.total, 0);
      const ticketPromedio = totalPedidos > 0 ? ingresoTotal / totalPedidos : 0;
      
      return {
        tienda: this.nombre,
        version: this.version,
        productos: totalProductos,
        usuarios: totalUsuarios,
        pedidos: totalPedidos,
        ingresoTotal: `${ingresoTotal.toFixed(2)} ${this.moneda}`,
        ticketPromedio: `${ticketPromedio.toFixed(2)} ${this.moneda}`,
        usuariosPorTipo: this.usuarios.reduce((acc, u) => {
          acc[u.tipo] = (acc[u.tipo] || 0) + 1;
          return acc;
        }, {})
      };
    }
  };
}
